Adopting the Robertson Example from the DifferentialEquation Docs https://diffeq.sciml.ai/stable/tutorials/advancedodeexample/#Handling-Mass-Matrices

function rober(du,u,p,t)
  y₁,y₂,y₃ = u
  du[1] = -0.04 * y₁ + 1e4 * y₂ * y₃
  du[2] =  0.04 * y₁ - 1e4 * y₂ * y₃ - 3e7 * y₂^2
  du[3] =  y₁ + y₂ + y₃ - 1
  nothing
end
