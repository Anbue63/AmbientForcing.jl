The coupling function is modeled by a differential algebraic equation with mass matrix 0 * de[1] = e[2] * sin(vs[1] - vd[1] + α) / N - e[1] is equivalent to e[1] = e[2] * sin(vs[1] - vd[1] + α) / N

function kuramoto_plastic_edge!(de, e, v_s, v_d, p, t)
    de[1] =  e[2] * sin(v_s[1] - v_d[1] + α) / N_plastic - e[1]
    de[2] = - ϵ * (sin(v_s[1] - v_d[1] + β) + e[2])

    nothing
end

function kuramoto_plastic_vertex!(dv, v, edges, p, t)
    dv .= 0
    for e in edges
        dv .-= e[1]
    end
end
